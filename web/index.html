01. Collections & Data

Type-safe database operations with a fluent API. Works in Node, Deno, and the Browser. 

import { RocketBaseClient } from "@vski/sdk";

// Initialize client
const client = new RocketBaseClient("https://api.vski.ai");

// Access typed collection
const users = client.collection("users");

// Create a new record
const newUser = await users.create({
    username: "neo",
    role: "hacker",
    active: true
});

// Query with advanced filtering
const activeHackers = await users.getList(1, 20, {
    filter: "role = 'hacker' && active = true",
    sort: "-created"
});

console.log(`Found ${activeHackers.totalItems} active hackers.`);

03. Durable Workflows

Write reliable, long-running workflows as code using the functional API. Automatic retries, durable sleep, and state recovery. 

import { workflow, step } from "@vski/sdk";

// Define individual steps
const lockSystem = step("initiate-lockdown", async (ctx) => {
    console.log("System locked.");
    return { lockedAt: new Date() };
});

const checkOverride = step("wait-for-auth", async (ctx) => {
    // Pauses execution until signal received
    const signal = await ctx.waitForSignal("admin-override");
    return signal.authorized;
});

// Compose the workflow
export const SystemPurge = workflow("system-purge")

SystemPurge.run(async (ctx) => {
    await lockSystem();

    // Durable sleep (survives worker restarts)
    await ctx.sleep("10s");

    const authorized = await checkOverride();
    if (!authorized) {
        return "Purge aborted.";
    }

    return "System purged.";
});
